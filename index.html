📋 Project 3: To-Do List

Add, show, and delete tasks using buttons

let tasks = [];
btnAdd.onclick = function() {
  tasks.push(inputEl.value);
  render();
};
function render() {
  listEl.innerHTML = "";
  tasks.forEach((t, i) => {
    let li = document.createElement("li");
    li.textContent = t;
    let btn = document.createElement("button");
    btn.textContent = "Delete";
    btn.onclick = () => {
      tasks.splice(i, 1);
      render();
    };
    li.appendChild(btn);
    listEl.appendChild(li);
  });
}

📘 Concepts Learned

  • Arrays: push, splice
  • Loops: forEach
  • DOM Manipulation: createElement, appendChild
  • Events inside loops

❓ Project 4: Quiz App

Show questions, accept true/false answers, calculate score

let questions = [
  { q: "HTML is a language", a: true },
  { q: "JS is only for games", a: false }
];
let current = 0, score = 0;
function showQ() {
  qEl.textContent = questions[current].q;
}
function check(ans) {
  if (ans === questions[current].a) score++;
  current++;
  current < questions.length ? showQ() : result();
}
